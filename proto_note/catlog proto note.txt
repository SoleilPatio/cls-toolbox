3. cat /proc/uptime
    import subprocess

    # 执行 adb shell cat /proc/uptime 命令
    result = subprocess.run(["adb", "shell", "cat", "/proc/uptime"], capture_output=True, text=True)

    # 获取输出并分割整数部分和小数部分
    output = result.stdout.strip().split()
    uptime_seconds = float(output[0])
    uptime_integer = int(uptime_seconds)
    uptime_decimal = uptime_seconds - uptime_integer

    print("Uptime in seconds:", uptime_seconds)
    print("Uptime integer part:", uptime_integer)
    print("Uptime decimal part:", uptime_decimal)
